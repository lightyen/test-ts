<variant> := <plain-variant> | <arbi-variant> | <any-variant> | <group-variant>

<plain-decl> := [!][-]<identifier>[/<modifier>][!]

<arbi-decl> := [!][-]<identifier>"-["<css-value>"]"[/<modifier>][!]

<modifier> := string | [ string ]

- *plain-variant*
